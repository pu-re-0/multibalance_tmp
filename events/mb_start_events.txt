country_event = {
	id = mb_start.2
	hide_window = yes
	trigger = {
		is_ai = no
		NOT = {
			has_global_flag = mb_fast_started
		}
	}
	is_triggered_only = yes
	immediate = {
		country_event = { id = mb_start.2 days = 360 }
		if = {
			limit = { end_game_years_passed >= 0 }
			country_event = { id = crisis_trigger.1 }
			set_global_flag = mb_fast_started
		}
	}
}
